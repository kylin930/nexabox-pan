NexaboxDrive API 接口文档

内部版本 | Web 1.0 兼容版 | v1.3 更新分享功能

一、 全局鉴权说明

所有私有接口请求均需要通过 HTTP Header 传递 Token 进行身份验证。公共分享接口无需此 Header。

Header 键 格式与说明
Authorization Bearer <您的Token>

二、 核心流程详细说明

1. 大文件切片上传步骤 (Upload Chunking)

系统采用前端切片直传 OSS 的方案：

  1. 计算切片： 按固定大小（250MB）划分分块。
  2. 分块伪装： 伪装成 image/png 格式以绕过限制。
  3. 循环上传： 获取 OSS 签名并 POST 二进制数据。
  4. 元数据提交： 上传完成后提交分块 URL 数组至 /api/files。

2. 分块合并与流式下载 (Download Merging)

下载时不进行物理合并，通过 Service Worker 按顺序拉取分块流式写入。说明见 sw.js。

3. 文件分享流程 (File Sharing) [新增]

允许将已上传的文件生成公共访问链接：

  1. 生成分享： 携带 Token 请求 /api/share，服务端生成唯一的 shareId。
  2. 访问链接： 用户访问 /share.html?id=SHARE_ID，此页面无需登录。
  3. 读取元数据： 分享页请求公共接口 /api/public/share 获取该文件的分块信息。
  4. 流式下载： 分享页通过 sw.js 启动与主站相同的分块拼接下载逻辑。

三、 接口列表

1. 获取文件列表 (List Files)

  • 接口路径： /api/files
  • 请求方式： GET
返回示例： 
[
  {
    "id": "FILE_1714000000000",
    "filename": "backup.tar.gz",
    "size": 1073741824,
    "chunks": ["https://oss-url/part1", "https://oss-url/part2"]
  }
]

2. 创建分享链接 (Create Share) [新增]

  • 接口路径： /api/share
  • 请求方式： POST
  • 鉴权： 需要 Token
请求参数： 
{
  "fileId": "FILE_1714000000000"
}
返回示例： 
{
  "success": true,
  "shareId": "a1b2c3d4",
  "url": "/share.html?id=a1b2c3d4"
}

3. 获取公共分享信息 (Get Public Share Info) [新增]

  • 接口路径： /api/public/share
  • 请求方式： GET
  • 鉴权： 无需 Token (公开)
URL 参数： ?id=a1b2c3d4
返回示例： 返回该文件的原始元数据（filename, size, chunks）。

四、 客户端接入示例 (C#)

以下示例基于 .NET 6+ 和 System.Net.Http 编写，展示了核心的上传和下载逻辑。

1. C# 切片上传示例 (Chunked Upload)

using System;
using System.IO;
using System.Net.Http;
using System.Net.Http.Headers;
using System.Text.Json;
using System.Collections.Generic;
using System.Threading.Tasks;

public class NexaboxUploader
{
    private static readonly HttpClient client = new HttpClient();
    private const string API_BASE = "https://drive.nexabox.de";
    private const string TOKEN = "您的TOKEN";
    private const int CHUNK_SIZE = 250 * 1024 * 1024; // 每个分块250MB

    public static async Task UploadFileAsync(string filePath)
    {
        client.DefaultRequestHeaders.Authorization = new AuthenticationHeaderValue("Bearer", TOKEN);
        
        FileInfo fileInfo = new FileInfo(filePath);
        string originalName = fileInfo.Name;
        long totalSize = fileInfo.Length;
        int totalChunks = (int)Math.Ceiling((double)totalSize / CHUNK_SIZE);
        List<string> chunkUrls = new List<string>();

        using (FileStream fs = new FileStream(filePath, FileMode.Open, FileAccess.Read))
        {
            for (int i = 0; i < totalChunks; i++)
            {
                Console.WriteLine($"准备上传分片 {i + 1}/{totalChunks}...");
                
                // 1. 获取签名
                var sigRes = await client.GetStringAsync($"{API_BASE}/api/signature");
                var sigData = JsonDocument.Parse(sigRes).RootElement;
                string ossHost = sigData.GetProperty("host").GetString();
                string timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds().ToString();
                
                // 构造伪装文件名与 OSS Key
                string fakeFileName = $"{originalName}.{i:D3}.png";
                string nameWithoutExt = Path.GetFileNameWithoutExtension(fakeFileName);
                string ossKey = $"{nameWithoutExt}-{timestamp}-{fakeFileName}";

                // 2. 读取当前切片数据
                int currentChunkSize = (int)Math.Min(CHUNK_SIZE, totalSize - (i * CHUNK_SIZE));
                byte[] buffer = new byte[currentChunkSize];
                await fs.ReadAsync(buffer, 0, currentChunkSize);

                // 3. 构建表单直传 OSS
                using (var content = new MultipartFormDataContent())
                {
                    content.Add(new StringContent(sigData.GetProperty("signature").GetString()), "Signature");
                    content.Add(new StringContent(sigData.GetProperty("accessKeyId").GetString()), "OSSAccessKeyId");
                    content.Add(new StringContent(sigData.GetProperty("policy").GetString()), "policy");
                    content.Add(new StringContent("200"), "success_action_status");
                    content.Add(new StringContent(ossKey), "key");

                    var fileContent = new ByteArrayContent(buffer);
                    fileContent.Headers.ContentType = MediaTypeHeaderValue.Parse("image/png");
                    content.Add(fileContent, "file", fakeFileName);

                    var uploadRes = await client.PostAsync(ossHost, content);
                    uploadRes.EnsureSuccessStatusCode();

                    string finalUrl = $"{ossHost}{ossKey}";
                    chunkUrls.Add(finalUrl);
                    Console.WriteLine($"分片 {i + 1} 上传成功: {finalUrl}");
                }
            }
        }

        // 4. 提交文件元数据
        var metaData = new
        {
            filename = originalName,
            size = totalSize,
            chunks = chunkUrls
        };
        var jsonContent = new StringContent(JsonSerializer.Serialize(metaData), System.Text.Encoding.UTF8, "application/json");
        var metaRes = await client.PostAsync($"{API_BASE}/api/files", jsonContent);
        
        if (metaRes.IsSuccessStatusCode)
            Console.WriteLine("文件元数据保存成功，上传完毕！");
    }
}

2. C# 流式合并下载示例 (Stream Download)

using System;
using System.IO;
using System.Net.Http;
using System.Collections.Generic;
using System.Threading.Tasks;

public class NexaboxDownloader
{
    private static readonly HttpClient client = new HttpClient();

    // 假设已从 /api/files 获取到了对应的 chunk 列表和文件名
    public static async Task DownloadFileAsync(string savePath, List<string> chunkUrls)
    {
        Console.WriteLine($"开始下载文件，共包含 {chunkUrls.Count} 个分片...");

        // 使用 Append 模式打开或创建文件流
        using (FileStream fs = new FileStream(savePath, FileMode.Append, FileAccess.Write, FileShare.None))
        {
            for (int i = 0; i < chunkUrls.Count; i++)
            {
                string chunkUrl = chunkUrls[i];
                Console.WriteLine($"正在拉取分片 {i + 1}/{chunkUrls.Count}...");

                // 以流的方式获取响应
                using (var response = await client.GetAsync(chunkUrl, HttpCompletionOption.ResponseHeadersRead))
                {
                    response.EnsureSuccessStatusCode();
                    using (var stream = await response.Content.ReadAsStreamAsync())
                    {
                        // 将分片流直接复制到本地文件流中
                        await stream.CopyToAsync(fs);
                    }
                }
                Console.WriteLine($"分片 {i + 1} 写入完成。");
            }
        }
        
        Console.WriteLine($"文件合并下载成功！保存至：{savePath}");
    }
}

// 示例：如何请求生成分享链接
public async Task<string> GenerateShareLink(string fileId)
{
    var content = new StringContent(JsonSerializer.Serialize(new { fileId }), Encoding.UTF8, "application/json");
    var response = await client.PostAsync($"{API_BASE}/api/share", content);
    var result = JsonDocument.Parse(await response.Content.ReadAsStringAsync()).RootElement;
    return result.GetProperty("url").GetString();
}
